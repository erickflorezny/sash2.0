    <script>
        const promptInput = document.getElementById('promptInput');
        const responseCard = document.getElementById('responseCard');
        const typingIndicator = document.getElementById('typingIndicator');
        const aiResponse = document.getElementById('aiResponse');

        // Auto-resize textarea
        promptInput.addEventListener('input', function() {
            this.style.height = 'auto';
            this.style.height = this.scrollHeight + 'px';
        });

        // Enter key handling
        promptInput.addEventListener('keypress', function(e) {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                askAI();
            }
        });

        const aiResponses = {
            windows: "I'd recommend our energy-efficient double-hung windows for your project. They feature Low-E glass, argon gas filling, and come with a lifetime warranty. Installation typically takes 1-2 days per room, and we offer 0% financing for qualified customers. Our windows can reduce your energy bills by up to 30%. Would you like to schedule a free consultation to see samples and get an exact quote?",
            
            bathroom: "Our luxury bathroom remodels are completed in just 5-7 days! We handle everything from design to final touches - new tiles, vanities, showers, and fixtures. Popular upgrades include walk-in showers, heated floors, and smart mirrors. Most projects range from $15,000-$35,000 depending on size and finishes. We provide 3D designs before starting so you can visualize your new space. Ready to transform your bathroom?",
            
            siding: "Our premium vinyl siding is perfect for durability and low maintenance. It resists weather, won't rot or warp, and comes in dozens of colors and styles. Installation typically takes 3-5 days for an average home. We also offer fiber cement and wood options. All siding includes insulation upgrades and lifetime warranties. Most homeowners see improved curb appeal and energy efficiency. What style of home do you have?",
            
            cost: "Our pricing is very competitive and includes everything - materials, labor, permits, and cleanup. Windows typically range $400-$800 per window installed. Bathroom remodels start around $15,000. Siding projects average $12,000-$25,000 for a full home. We offer free estimates, 0% financing, and lifetime warranties. Every project includes a detailed quote with no hidden fees. Would you like to schedule a free consultation?",
            
            financing: "We offer excellent financing options! Qualified customers can get 0% APR for up to 12 months, or extended terms up to 10 years with competitive rates. We work with multiple lenders to find the best option for your budget. Most approvals are instant, and you can use financing for any of our services. Our team handles all the paperwork. Would you like to check your financing options?",
            
            timeline: "Here are our typical project timelines: Windows take 1-2 days per room, bathroom remodels are completed in 5-7 days, and full home siding takes 3-5 days. We schedule around your convenience and provide exact start dates. Our crews are fully licensed, insured, and professional. We clean up completely each day and protect your landscaping. Weather can sometimes extend outdoor projects by a day or two.",
            
            default: "I'd be happy to help with your home improvement project! We specialize in premium windows, luxury bathroom remodels, and durable siding installations. All our work comes with lifetime warranties and we offer 0% financing for qualified customers. Our expert team has been serving homeowners for over 20 years with thousands of satisfied customers. What specific questions do you have about your project?"
        };

        function askAI() {
            const question = promptInput.value.trim().toLowerCase();
            if (!question) {
                promptInput.focus();
                return;
            }

            responseCard.classList.add('active');
            typingIndicator.style.display = 'flex';
            aiResponse.classList.remove('show');

            // Smooth scroll to response
            setTimeout(() => {
                responseCard.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
            }, 100);

            // Simulate AI processing time
            setTimeout(() => {
                typingIndicator.style.display = 'none';
                
                let response = aiResponses.default;
                
                if (question.includes('window')) response = aiResponses.windows;
                else if (question.includes('bath') || question.includes('shower')) response = aiResponses.bathroom;
                else if (question.includes('siding') || question.includes('exterior')) response = aiResponses.siding;
                else if (question.includes('cost') || question.includes('price') || question.includes('money')) response = aiResponses.cost;
                else if (question.includes('financing') || question.includes('payment')) response = aiResponses.financing;
                else if (question.includes('time') || question.includes('long') || question.includes('schedule')) response = aiResponses.timeline;

                aiResponse.textContent = response;
                aiResponse.classList.add('show');
            }, 2000);
        }

        // Demo interaction after page load
        setTimeout(() => {
            promptInput.placeholder = "Try asking: How much do new windows cost?";
        }, 3000);

        setTimeout(() => {
            promptInput.placeholder = "Or ask: What's included in a bathroom remodel?";
        }, 6000);

        setTimeout(() => {
            promptInput.placeholder = "What home improvement project can we help you with today?";
        }, 9000);
    </script>
